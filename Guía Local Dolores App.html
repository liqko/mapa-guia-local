✅ 1. Componente React (MapaLocales.js)

Este es el componente para agregar a tu app Guía Local Dolores:

jsx
// MapaLocales.js
import React, { useEffect } from 'react';
import L from 'leaflet';
import 'leaflet/dist/leaflet.css';

const locales = [
  {
    nombre: "Panadería La Dulce",
    lat: -36.312345,
    lng: -57.679876,
    rubro: "Panadería",
    direccion: "Belgrano 123",
    web: "https://panaderialadulce.com"
  },
  {
    nombre: "Hotel Dolores",
    lat: -36.311111,
    lng: -57.682222,
    rubro: "Hotel",
    direccion: "San Martín 456",
    web: "https://hoteldolores.com"
  }
];

const MapaLocales = () => {
  useEffect(() => {
    const map = L.map('map').setView([-36.3123, -57.6799], 15);

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '&copy; OpenStreetMap contributors'
    }).addTo(map);

    locales.forEach(local => {
      L.marker([local.lat, local.lng])
        .addTo(map)
        .bindPopup(`<strong>${local.nombre}</strong><br>${local.rubro}<br>${local.direccion}<br><a href="${local.web}" target="_blank">Ver sitio</a>`);
    });
  }, []);

  return (
    <div style={{ height: '100vh' }}>
      <div id="map" style={{ height: '100%', width: '100%' }}></div>
    </div>
  );
};

export default MapaLocales;
